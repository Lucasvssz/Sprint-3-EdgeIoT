//
int sensor = 0;

void setup()
{
  pinMode(A0, INPUT);
  pinMode(8, OUTPUT);
  Serial.begin(9600);
}

void loop()
{
  sensor = analogRead(A0);
  if (sensor > 300) {
    digitalWrite(8, HIGH);
  } else {
    digitalWrite(8, LOW);
  }
  Serial.print("{"sensor": ");
  Serial.print(sensor);
  Serial.println("}");

  delay(100); // Wait for 100 millisecond(s)
}
